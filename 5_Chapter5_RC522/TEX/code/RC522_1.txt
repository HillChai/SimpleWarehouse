#include <SPI.h>
#include <MFRC522.h>
 //引脚11，12，13已经被RC522的MOSI，MISO，SCK所使用
 //除了引脚11，12，13外的任意空闲数字引脚皆可使用
 //定义为引脚9和10为常见布局
#define RST_PIN         9           
#define SS_PIN          10          
// 创建MFRC522实例
MFRC522 mfrc522(SS_PIN, RST_PIN);   
//定义全局变量，由串口通信选择读或写
char choice;

void setup() {
  // 初始化PC端的串口通信
  Serial.begin(9600);      
  //初始化SPI总线                
  SPI.begin();                   
  //初始化MRFC522卡                       
  mfrc522.PCD_Init();    
  // 向串口发送消息选择读或写
  Serial.println("1.Read card \n2.Write to card \n3. ... \n");
}
